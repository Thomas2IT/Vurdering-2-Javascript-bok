Til de som lurer på hvordan man vanligvis bruker ctx kode, så er dette et fint eksempel hvor jeg lager to parallele streker: 

        
        "canvas id="mitteksempel" width="300" height="150" style="border:1px solid #d3d3d3;"
        
        var tegneflateeksempel = document.getElementById("mitteksempel");
        var ctx = tegneflateeksempel.getContext("2d");
        
        ctx.beginPath();              
        ctx.lineWidth = "5";
        ctx.strokeStyle = "Yellow";  
        ctx.moveTo(0, 75);
        ctx.lineTo(200, 75);
        ctx.stroke();  
        
        ctx.beginPath();
        ctx.strokeStyle = "Blue";  
        ctx.moveTo(0, 100);
        ctx.lineTo(300, 100);            
        ctx.stroke();

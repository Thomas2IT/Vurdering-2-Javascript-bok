    function C0ntinue() {
        var text = "";
        var antall;
        for (antall = 0; antall < 5; antall++) {
          if (antall === 3) {
            continue;
          }
          text += "Nummer " + antall + "
";
        }
        document.getElementById("demo").innerHTML = text;
